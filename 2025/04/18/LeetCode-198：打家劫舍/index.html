publicclassSolution {
    publicintrob(int[] nums) {
        intn= nums.length;
        if (n == 0) return0;
        if (n == 1) return nums[0];

        int[] dp = newint[n];
        dp[0] = nums[0];
        dp[1] = Math.max(nums[0], nums[1]);

        for (inti=2; i < n; i++) {
            dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
        }

        return dp[n - 1];
    }
}
